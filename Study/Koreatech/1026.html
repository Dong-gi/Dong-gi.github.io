import java.util.*;

public class Main {

    private static char[] cs;
    private static boolean[] bs;
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int testCase = scanner.nextInt();
        while(testCase-- > 0) {
            cs = scanner.next().toCharArray();
            bs = new boolean[cs.length]; // true means "removed"
            
            for(int i = 1; i < cs.length; ++i) {
                if(!bs[i]) {
                    int left = getLeftMost(i);
                    while(left >= 0 && i < cs.length && cs[left] == cs[i]) {
                        bs[left] = bs[i++] = true;
                        left = getLeftMost(left);
                    }
                }
            }
            
            StringBuilder builder = new StringBuilder();
            for(int i = 0; i < bs.length; ++i) {
                if(!bs[i]) {
                    builder.append(cs[i]);
                }
            }
            System.out.println(builder.toString());
        }
    }
    
    /* Find leftmost char's index from (idx-1)
     * If there's no char, return -1
     */
    private static int getLeftMost(int idx) {
        for(int i = idx-1; i >= 0; --i) {
            if(!bs[i]) {
                return i;
            }
        }
        return -1;
    }

}
