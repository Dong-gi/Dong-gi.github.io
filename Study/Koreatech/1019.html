import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int testCase = scanner.nextInt();
        while(testCase-- > 0) {
            String numStr = scanner.next();
            char[] nums = numStr.toCharArray();
            ArrayList<Character> big = new ArrayList<>();
            for(char c : nums) {
                big.add(c);
            }
            ArrayList<Character> small = new ArrayList<>();
            small.addAll(big);
            big.sort((Character c1, Character c2) -> c2.compareTo(c1));
            small.sort((Character c1, Character c2) -> c1.compareTo(c2));

            int bigInt = Integer.parseInt(numStr);
            int smallInt = Integer.parseInt(numStr);

            for(int i = 0; i < nums.length; ++i) {
                char c = big.remove(0); 
                if(c > nums[i]) {
                    StringBuilder builder = new StringBuilder(numStr);
                    builder.setCharAt(i, c);
                    builder.setCharAt(numStr.lastIndexOf(c), nums[i]);
                    bigInt = Integer.parseInt(builder.toString());
                    break;
                }
            }

            if(small.get(0) == '0') {
                char c = '0';
                for(int i = 1; i < nums.length; ++i) {
                    if(c < nums[i]) {
                        StringBuilder builder = new StringBuilder(numStr);
                        builder.setCharAt(i, c);
                        builder.setCharAt(numStr.lastIndexOf(c), nums[i]);
                        smallInt = Integer.parseInt(builder.toString());
                        break;
                    }
                }
                try {
                    while(small.get(0) == '0') {
                        small.remove(0);
                    }
                } catch(Exception e) {}
            }
            try {
                for(int i = 0; i < nums.length; ++i) {
                    while(nums[i++] == '0');
                    char c = small.remove(0);
                    if(c < nums[--i]) {
                        StringBuilder builder = new StringBuilder(numStr);
                        builder.setCharAt(i, c);
                        builder.setCharAt(numStr.lastIndexOf(c), nums[i]);
                        smallInt = (smallInt < Integer.parseInt(builder.toString()))?
                                smallInt : Integer.parseInt(builder.toString());
                    }
                }
            } catch(Exception e) {}
            //System.out.println(smallInt + " " + bigInt);
            System.out.println(bigInt - smallInt);
        }
    }

}
