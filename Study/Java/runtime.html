public class Test{
	public static void main(String[] args) {
		Runtime r = Runtime.getRuntime();
		
		r.addShutdownHook(new Thread(()-> {System.out.println("Good Bye!");}));
		// r.exec() -> process 참고
		// r.load() : dll에 대한 full path 포함한 파일명을 받는다.
		// r.loadLibrary() -> JNI 참고
		
		System.out.println(r.totalMemory());
		for(int i = 0; i < 200; ++i) new Thread();
		System.out.println(r.freeMemory()); // 가용 메모리 추정 크기
		r.gc(); // GC 수행. 비교. runFinalization() 사용하지 않은 객체 정리
		System.out.println(r.freeMemory());
		
		r.exit(0); // 비교. halt() 즉시 종료
		System.out.println("Doesn't reach here");
	}
}
