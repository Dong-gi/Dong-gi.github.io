import java.util.*;

public class Test {
	public static void main(String[] args) {
		System.out.println(Float.MAX_EXPONENT);
		System.out.println(Float.MAX_VALUE);
		System.out.println(Float.NaN); // 0/0.
		System.out.println(Double.POSITIVE_INFINITY); // +/0.
		
		System.out.println(Float.floatToIntBits(1.23F));
		System.out.println(Double.doubleToLongBits(1.23));
		
		System.out.println(Float.intBitsToFloat(123));
		System.out.println(Double.longBitsToDouble(123456L));
		
		System.out.println(Float.isFinite(1.23F));
		
		System.out.println(Double.parseDouble("1.234"));
		System.out.println(Double.toHexString(1.234));
	}
}
