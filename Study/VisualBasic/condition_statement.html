Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim n As Integer = 10
        If n Mod 2 = 0 Then
            TextBox1.Text = "짝수"
            If n < 10 Then
                TextBox1.Text &= ", 10보다 작음"
            ElseIf n < 20 Then
                TextBox1.Text &= ", 20보다 작음"
            End If
        Else
            TextBox1.Text = "홀수"
        End If
    End Sub

    Private Sub Button2_Click(sender As Object, e As EventArgs) Handles Button2.Click
        Dim n As Integer = 10
        If n Mod 2 = 0 Then TextBox1.Text = "짝수"
        ' Else를 못씀...
    End Sub

    Private Sub Button3_Click(sender As Object, e As EventArgs) Handles Button3.Click
        Dim str As String = "Hello"
        Select Case str
            Case "Hello" ' break 키워드 없이 한 케이스만 실행하고 종료
                TextBox1.Text = "H"
            Case "hello", "world" ' 여러 개 가능
                TextBox1.Text = "h"
            Case Else
                TextBox1.Text = "뭔지 모르겠음"
        End Select
    End Sub

    Private Sub Button4_Click(sender As Object, e As EventArgs) Handles Button4.Click
        Dim n As Integer = 10
        Select Case n Mod 2
            Case Is = 0
                TextBox1.Text = "짝수"
            Case Is > 0
                TextBox1.Text = "홀수"
            Case Else
                TextBox1.Text = "음수"
        End Select
    End Sub

End Class
