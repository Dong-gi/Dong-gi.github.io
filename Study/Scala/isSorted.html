/**
 * Series.scala
 */
object Series {
  def isSorted[A](arr: Array[A], compare: (A, A) => Boolean): Boolean = {
    def sorted(i: Int): Boolean = {
      if(i >= arr.length) { true }
      else if(compare(arr(i-1), arr(i))) { sorted(i+1) }
      else { false }
    }
    sorted(1)
  }
}
