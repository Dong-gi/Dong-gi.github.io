class Name { // typeof Name === "function"
    constructor() {}

    static func() {}

    func2() {
        // 표기 "Name.prototype.func" = "Name#func"
    }
}

const n = new Name();
n.__proto__.func2 === n.func2;

class Main {}
class Sub extends Main {}


class MixIn {
    // 다른 클래스 객체에 특성 주입
    static inject(obj) {
        obj.func = function() {};
    }
}
