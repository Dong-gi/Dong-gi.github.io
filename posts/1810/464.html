3. python과 비슷한 기능을 하는 Destructuring Assignment
const obj = { a:1, b:2 };
let {a, b, c} = obj;
console.log([a, b, c]);
console.log(({c, b, a} = obj) === obj);

function doSomething({a, b, arr}) {
    console.log([a, b, arr]);
}

let arr = [1, 2, 3, 4, 5];
let a, b;
[a, b, ...arr] = arr;
[a, b] = [b, a];
doSomething({a: a, b: b, arr: arr});
