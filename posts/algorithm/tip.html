  2. d = 22 # 22일
  3. m = 12 # 1(3월), 2(4월), ..., 12(2월)
  4. c = 20 # year // 100
  5. y = 19 # (year % 100), 1-2월이면 -1
  6. w = (d + floor(2.6 * m - 0.2) - 2 * c + y + (y // 4) + (c // 4)) % 7
