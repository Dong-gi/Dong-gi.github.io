1. 어노테이션을 반복적으로 정의할 수 있음을 표현.
예. https://docs.oracle.com/javase/tutorial/java/annotations/repeating.html
import java.lang.annotation.Repeatable;

@Repeatable(Schedules.class)
public @interface Schedule {
    Schedule[] value(); // 여러 어노테이션 값을 유지하기 위해 꼭 필요
    String dayOfMonth() default "first";
    String dayOfWeek() default "Mon";
    int hour() default 12;
}
