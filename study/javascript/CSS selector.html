<form id="form1">
    <select name="colorSelection">
        <option value="red">Red</option>
        <option value="green">Green</option>
        <option value="blue">Blue</option>
    </select>
</form>

<script>
document.querySelector('option[value="red"]').style.color = "red";
document.querySelector('option[value="green"]').style.color = "green";
document.querySelector('option[value="blue"]').style.color = "blue";
// querySelector('CSS 선택자')
document.querySelector('form[id="form1"]').colorSelection.onchange = function() {
    document.getElementById('p1').style.color = document.getElementById('form1').colorSelection.value;
};
</script>
