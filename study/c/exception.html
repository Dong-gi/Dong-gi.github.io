#include <iostream>

using namespace std;

int main(void)
{
	try {
		throw exception("Hello");
	}
	catch (const exception& e) {
		cout << e.what();
	}

	try {
		throw new exception("Hello");
	}
	catch (const exception* e) {
		cout << e->what();
		delete e;
	}
	
	int pls_dont_end;
	cin >> pls_dont_end;
	return 0;
}
