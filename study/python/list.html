'''
리스트
'''

print('리스트 생성 1 : []')
list1 = [1, 2, 3, 4, 5]
print(list1)


print('리스트 생성 2 : list()')
list1 = list([2, 3, 4, 5, 6])
print(list1)

list1 = list(['hello world'])
print(list1)

list1 = list('hello world')
print(list1)

list1 = list(('world hello'))
print(list1)


print('리스트 요소 추출')
print(list1[3])
print(list1[::-1])


print('리스트 병합')
list1.extend(('python'))
print(list1)

list1 += ['nohtyp']
print(list1)


print('리스트 뒤에 추가')
list1.append('append')
print(list1)


print('리스트 중간에 추가')
list1.insert(3, 'third')
print(list1)


print('리스트 요소 삭제')
list1.remove('l')
print(list1)

del list1[0]
del list1[-1]
print(list1)

print(list1.pop())
print(list1.pop(0))


print('리스트 요소 위치 확인')
print(list1.index('l'))


print('리스트 요소 포함 여부 확인')
print('python' in list1)


print('리스트 요소 개수 확인')
print(list1.count('l'))
print(len(list1))


print('리스트 정렬')
list1.sort()
print(list1)
list1.sort(reverse=True)
print(list1)


print('복사')
list2 = list1 # 얕은 복사
list3 = list1.copy() # 깊은 복사
list4 = list(list1) # 깊은 복사
list5 = list1[::] # 깊은 복사
list1[1] = 'First'
print(list2)
print(list3)
print(list4)
print(list5)
