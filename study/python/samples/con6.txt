>>> # 튜플 생성
>>> tuple1 = (1, 2, 3, 'abcd')
>>> tuple1
(1, 2, 3, 'abcd')
>>> 
>>> tuple1 = tuple('h' * 3)
>>> tuple1 += (4, ) # 하나일 땐 뒤에 콤마를 붙여야 튜플로 인식한다.
>>> tuple1 += (5, 6)
>>> tuple1
('h', 'h', 'h', 4, 5, 6)
>>> 
>>> # 튜플 언패킹
>>> _, _, _, a, b, c = tuple1
>>> c, b, a = a, b, c
>>> a, b, c
(6, 5, 4)
