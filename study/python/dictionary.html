'''
딕셔너리
'''

print('딕셔너리 : "키-값"쌍 저장')

print('생성')
dic = {
    'key' : 'value',
    'hello' : 'world'
}
print(dic)

dic = dict()
dic['key1'] = 'val1';
dic['pine'] = 'apple';
print(dic)
dic['key1'] = 'val2';
print(dic)


print('형 변환')
print(dict([['a', 'b'], ['c', 'd']]))
# print(dict([['a', 'b', 'c', 'd']])) # error
print(dict([('a', 'b'), ('c', 'd')]))
# print(dict([('a', 'b', 'c', 'd')])) # error
print(dict([['a', 'b'], ('c', 'd')]))
print(dict(['ab', 'cd', 'ef']))
print(dict(('ab', 'cd', 'ef')))


print('병합 : 기존 값은 갱신')
dic.update({
    'key1' : 'val3',
    'key2' : 'val4'
})
print(dic)


print('요소 삭제')
del dic['key1']
print(dic)

dic.clear()
print(dic)


print('키 포함 여부')
print('key2' in dic)


print('모든 키 얻기')
dic = {
    'a' : 'b',
    'c' : 'd'
}
print(dic.keys()) # 리스트
print(type(dic.keys()))


print('모든 값 얻기')
print(dic.values()) # 리스트
print(type(dic.values()))


print('모든 "키-값"쌍 얻기')
print(dic.items()) # 튜플의 리스트
print(type(dic.items()))


print('복사')
dic2 = dic # 얕은 복사
dic3 = dic.copy() # 깊은 복사
dic['a'] = 'z'
print(dic2)
print(dic3)
