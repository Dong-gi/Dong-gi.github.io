print('내부함수 ~ 클로저')
def func1(name):
    def func2():
        print(name)
    return func2

func3 = func1('wiz')
func3()


print('람다함수')
func4 = lambda name : print(name)
func4('rowiz')


print('Generator')
def even_gen(max = 10):
    num = 2
    while num <= max:
        yield num
        num += 2

for n in even_gen():
    print(n)


print('Decorator : 함수에 동적으로 기능 추가')
def func5(func):
    def func6(name):
        print('Additional Codes...')
        return func(name)
    return func6

func7 = func5(func4)
func7('wiz')

@func5
def print_name(name):
    print(name)

print_name('wiz')
