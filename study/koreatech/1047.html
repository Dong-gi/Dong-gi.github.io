import java.util.HashSet;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int size = in.nextInt();
        in.nextLine();
        HashSet<Integer> set = new HashSet<Integer>();
        for( ; size>0; --size)
            set.add(in.nextInt());
        System.out.println(set.size());
    }
}
