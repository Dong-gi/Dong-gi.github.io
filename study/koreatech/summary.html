public class Summary {

    public static void main(String[] args) {

        System.out.println("문자열 뒤집기. reverse a string");

        String original = "reverse a string";
        String reverse = new StringBuilder(original).reverse().toString();
        System.out.println(original + "|" + reverse);


        System.out.println("소수 구하기. find primes");
        System.out.println("메모리만 허용된다면, 배열을 크게 만들어놓고 배수를 지워나가는 방식이 더 빠르다");
        findPrimesTo(10000000);

    }

    private static void findPrimesTo(int size) {
        boolean[] isPrime = new boolean[size+1];
        for (int i = 2; i <= size; i++) {
            isPrime[i] = true;
        }
        for (int i = 3; i*i <= size; i += 2) {
            try {
                if (isPrime[i]) {
                    for (int j = i; i*j <= size; j += 2) {
                        isPrime[i*j] = false;
                    }
                }
            } catch(Exception e) {}
        }
        int count = 1; // 2
        for (int i = 3; i <= size; i += 2) {
            if(isPrime[i]) {
                count += 1;
            }
        }
        System.out.println("Number of primes : " + count);
    }

}
