sealed trait List[+A]
case object Nil extends List[Nothing]
case class Cons[+A](head: A, tail: List[A]) extends List[A]

object List {
  // z(zero)는 빈 리스트에 대한 기본값
  /*
  def foldRight[A, B](l: List[A], z: B)(f: (A, B) => B): B = l match {
    case Nil => z
    case Cons(head, tail) => f(head, foldRight(tail, z)(f))
  }
  */
  def foldRight[A, B](l: List[A], z: B)(f: (A, B) => B): B = newFoldRight(l, z)(f)
  
  @annotation.tailrec
  def foldLeft[A, B](l: List[A], z: B)(f: (B, A) => B): B = l match {
    case Nil => z
    case Cons(head, tail) => foldLeft(tail, f(z, head))(f)
  }
  
  def sumR(l: List[Int]): Int = foldRight(l, 0)((x, y) => x + y)
  def sumL(l: List[Int]): Int = foldLeft(l, 0)(_ + _)
  
  def productR(l: List[Int]): Int = foldRight(l, 1)(_ * _)
  def productL(l: List[Int]): Int = foldLeft(l, 1)(_ * _)
  
  def lengthR[A](l: List[A]): Int = foldRight(l, 0)((a, x) => x + 1)
  def lengthL[A](l: List[A]): Int = foldLeft(l, 0)((x, a) => x + 1)
  
  def reverseR[A](l: List[A]): List[A] = foldRight(l, Nil: List[A])((a, l) => append(l, a))   
  def reverseL[A](l: List[A]): List[A] = foldLeft(l, Nil: List[A])((l, a) => Cons(a, l))
  
  def append[A](l: List[A], as: A*): List[A] = append(l, List(as: _*))
  def append[A](l1: List[A], l2: List[A]): List[A] = foldRight(l1, l2)((a, l) => Cons(a, l))
  
  def reduce[A](l: List[List[A]]): List[A] = foldLeft(l, Nil: List[A])((l1, l2) => append(l1, l2))
  
  def plus1(l: List[Int]): List[Int] = foldLeft(l, Nil: List[Int])((l, a) => append(l, a+1))
  
  // 목록의 구조를 유지하면서 각 요소를 수정
  // def map[A, B](l: List[A])(f: A => B): List[B] = foldLeft(l, Nil: List[B])((l, a) => append(l, f(a)))
  def map[A, B](l: List[A])(f: A => B): List[B] = flatMap(l)(f andThen ((B) => List(B)))
  
  // 조건을 만족하는 것만 유지
  def filter[A](l: List[A])(f: A => Boolean): List[A] = foldLeft(l, Nil: List[A])((l, a) => {
    if(f(a)) { append(l, a) }
    else { l }
  })
  
  // 수정한 목록들을 하나의 목록으로 이어서 작성
  def flatMap[A, B](l: List[A])(f: A => List[B]): List[B] = foldLeft(l, Nil: List[B])((l, a) => append(l, f(a)))
  
  def newFoldRight[A, B](l: List[A], z: B)(f: (A, B) => B): B = foldLeft(reverseL(l), z)((b, a) => f(a, b))
  
  def sumList(l1: List[Int], l2: List[Int]): List[Int] = (l1, l2) match {
    case (Nil, _) => Nil
    case (_, Nil) => Nil
    case (Cons(h1, t1), Cons(h2, t2)) => Cons(h1+h2, sumList(t1, t2))
  }
  
  def zipWith[A, B, C](l1: List[A], l2: List[B])(f: (A, B) => C): List[C] = (l1, l2) match {
    case (Nil, _) => Nil
    case (_, Nil) => Nil
    case (Cons(h1, t1), Cons(h2, t2)) => Cons(f(h1, h2), zipWith(t1, t2)(f))
  }
  
  def toString[A](l: List[A]): String = l match {
    case Nil => ""
    case Cons(head, tail) => head + " " + toString(tail)
  }
  
  def apply[A](as: A*): List[A] = {
    if(as.isEmpty) Nil
    else Cons(as.head, apply(as.tail: _*))
  }
}

import List._

object Main {
  def main(args: Array[String]): Unit = {
    var l = List(1, 2, 3, 4, 5)
    println(List.sumL(l))
    println(List.sumR(l))
    println(List.productL(l))
    println(List.productR(l))
    println(List.lengthL(l))
    println(List.lengthR(l))
    println(List.toString(List.reverseL(l)))
    println(List.toString(List.reverseR(l)))
    println(List.toString(List.reduce(List(List(1, 2, 3), List(4, 5, 6), List(7, 8, 9)))))
    println(List.toString(List.plus1(l)))
    println(List.toString(l)) // 원본이 그대로임을 확인하기 위해 출력
    println(List.toString(List.map(l)((i) => "Hello " + i + ",")))
    println(List.toString(List.filter(l)((i) => i % 2 == 0)))
    println(List.toString(List.flatMap(l)((i) => List(i, i, i))))
  }
}
