/**
 * Series.scala
 */
object Series {
  def fibonacci(n: Int): Int = {
    def next(n: Int, prev: Int, cur: Int): Int = {
        if(n < 1) { cur }
        else { next(n-1, cur, prev+cur) }
    }
    if(n < 2) { n }
    else { next(n-2, 1, 1) }
  }
}

/**
 * Main.scala
 */
 
 import Series._
 
 object Main {
  def main(args: Array[String]): Unit = {
    for(i <- 0 to 20) {
      print(fibonacci(i) + " ")
    }
  }
}
