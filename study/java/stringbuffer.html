import java.util.*;

public class Test {
	public static void main(String[] args) {
		StringBuffer b = new StringBuffer(10); // capacity
		b = new StringBuffer("Hello"); // CharSequence
		
		System.out.println(b.capacity());
		
		System.out.println(b.length());
		b.setLength(10);
		System.out.println(b.length());
		System.out.println(b);
		
		b.setCharAt(8, 'I');
		System.out.println(b);
		
		b.append(1).append(0.123).append(true).append('b').append(b);
		System.out.println(b);
		
		b.insert(b.length()/2, " ");
		System.out.println(b);
		
		System.out.println(b.reverse());
		System.out.println(b); // 원본 변화 && 반환
		
		b.delete(b.length()/2, b.length());
		System.out.println(b);
	}
}
