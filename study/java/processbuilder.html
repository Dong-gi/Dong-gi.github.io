import java.util.Map.Entry;

public class Test{
	public static void main(String[] args) {
		ProcessBuilder b = new ProcessBuilder();
		b.command("notepad"); // 메서드 get/set 구분이 없다.
		// b.directory() // 실행 위치 설정/반환
		// b.redirect~() // 스트림 조정
		b.redirectErrorStream(true); // process 입장에서, stdout을 에러 스트림으로도 이용
		
		for(Entry<String, String> e : b.environment().entrySet()) {
			System.out.printf("Key : %s, Val : %s\n", e.getKey(), e.getValue());
		}
		
		try {
			b.start();
		} catch (Exception e1) { e1.printStackTrace(); }
	}
}
