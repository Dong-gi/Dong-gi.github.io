import java.util.*;

public class Test {
	public static void main(String[] args) {
		System.out.println(Character.isDefined('햙')); // Unicode
		System.out.println(Character.isDigit('2'));
		System.out.println(Character.isIdentifierIgnorable('.'));
		System.out.println(Character.isJavaIdentifierStart('0'));
		System.out.println(Character.isJavaIdentifierPart('0'));
		System.out.println(Character.isISOControl('\t'));
		System.out.println(Character.isLetter('풻'));
		System.out.println(Character.isLowerCase('a'));
		System.out.println(Character.toUpperCase('a'));
		System.out.println(Integer.toHexString(Character.MAX_CODE_POINT));
		System.out.println(Character.charCount(0x100E00)); // 출력에 필요한 char 수
		System.out.println(Character.toChars(0x100E00));
		System.out.println(Character.isMirrored('가')); // 우->좌로 출력되는 문자
		System.out.println(Character.isWhitespace('\n'));
	}
}
