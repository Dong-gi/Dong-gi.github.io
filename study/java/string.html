import java.util.*;

public class Test {
	public static void main(String[] args) {
		System.out.println(new String("Hello World".toCharArray()));
		System.out.println(new String("Hello World".getBytes()));
		System.out.println(new String(new StringBuilder("Hello")));
		
		char[] chars = new char[10];
		"Hello World".getChars(3, 8, chars, 0);
		System.out.println(new String(chars));
		
		System.out.println("HELLO".equalsIgnoreCase("hello"));
		System.out.println("ABCD".regionMatches(true, 1, "bcd", 0, 3));
		
		System.out.println("Hello".startsWith("H"));
		System.out.println("Hello".endsWith("o"));
		System.out.println("  Hello  ".trim());
		
		System.out.printf("%x\n", "Hello".codePointAt(2));
	}
}
